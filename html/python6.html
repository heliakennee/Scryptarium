    def comp(motif1,motif2):
        n=0
        seq=""
        while motif1 not in seq and motif2 not in seq:
            n+=1
            seq+=choice(('P','F'))
        if motif1 in seq:
            return 1
        else:
            return 0

    def prob(k):
        s=0
        for i in range(k):
                s+=comp("PPF","FPP")
        return str(100-s*100/k)+'%',str(s*100/k)+'%'
