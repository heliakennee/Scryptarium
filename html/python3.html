    from math import *

    def estpremier(n):
        return len([i for i in range(1,int(sqrt(n))+1) if n%i == 0 and n != 1]) == 1

    def prem_recursif(n,index=1,lst=[]):
        if index > sqrt(n):
            return len(lst) == 1
        elif n%index == 0 and n != 1:
            return prem_recursif(n,index+1,[index]+lst)
        return prem_recursif(n,index+1,lst)

    def decomposer(n,index=2):
        if n == index :
            return [index]
        elif n%index==0 and prem_recursif(index):
            return [index] + decomposer(n//index,index)

Algortithme d'euclide


    def euclide(l,f):
        assert l != 0 and f != 0,"Attention: risque de division par 0"
        maxi = max(l,f)
        mini = min(l,f) 
        if maxi % mini == 0:
            return int(mini)
        else :
            return euclide(mini,maxi%mini)

Console :


    euclide(25,15)
    5

Algorithme de soustraction


    def soustraction(l,f):
        if l-f == 0:
            return l
        else :
            return soustraction(abs(l-f),min(l,f))
