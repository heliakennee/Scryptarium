    function convertirMonnaie($montant, $deviseOrigine) {
        $montantConverti = 0;
        define("taux", 1.10);
    
        if ($deviseOrigine == "EUR") {
            $montantConverti = $montant * taux;
        } elseif ($deviseOrigine == "USD") {
            $montantConverti = $montant / taux;
    
        }
        return $montantConverti;
    }
    
    do {
        echo "Veuillez entrer le montant à convertir : ";
        $montant = (float) trim(fgets(STDIN));
    
        echo "Veuillez entrer la devise d'origine (EUR ou USD) : ";
        $deviseOrigine = trim(fgets(STDIN));
    
        $resultat = convertirMonnaie($montant, $deviseOrigine);
        echo "Résultat de la conversion : " . $resultat . "\n";
    
      echo "Voulez-vous effectuer une autre conversion ? (O/N) : ";
      $reponse=trim(fgets(STDIN));
    
    } while ($reponse == "O");
    
    echo "Fin du programme. Merci d'avoir utilisé le convertisseur !";

Console :

    Fin du programme. Merci d'avoir utilisé le convertisseur !
