    from turtle import *
    from random import *

    right(60)

    def triangle(c):
        colormode(255)
        fillcolor(randint(0,255),randint(0,255),randint(0,255))
        begin_fill()
        for i in range(3):
            forward(c)
            right(120)
        end_fill()

    def ligne(c,n):
        if n > 0:
            triangle(c)
            forward(50)
            return ligne(c,n-1)
            
    def pyramide(c,n):
        if n > 0:
            ligne(c,n)
            right(120)
            forward(50)
            return pyramide(c,n-1)
