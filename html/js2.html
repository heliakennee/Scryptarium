Utilisations des listes

Change un nombre en liste

Transforme nombre en liste



    function liste(n) {
        if (Math.trunc(n/10) == 0) {
            return [n];
        } else {
            return liste(Math.trunc(n/10)).concat([n % 10]);
        }
    }

Console :


    liste(155);
    [1,2,3]

Somme des termes d'un nombre au carr√©



    function somme(n) {
        if (n<10) {
            return n ** 2;
        } else {
            return (n % 10) ** 2 + somme(Math.trunc(n / 10));
        }
    }

Console :

    14

Nombres heureux



    function heureux(n) {
        try {
            if (somme(n) == 1) {
                return true;
            } else {
                return heureux(somme(n));
            }
        } catch {
            return false
        }
    }

Algorithme de syracuse



    function syracuse(u) {
        if (u == 1) {
            return [1];
        } else {
            if (u%2==0) {
                u=Math.trunc(u/2);
            } else {
                u=3*u+1;
            }
            return [u].concat(syracuse(u));
        }
    }
