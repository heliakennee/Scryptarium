    $nbVals = intval(readline("Combien de valeurs voulez-vous saisir ? "));
    $vals = array();
    $pos = 0;
    $neg = 0;
    
    for ($i = 0; $i < $nbVals; $i++) {
        $vals[] = intval(readline("Saisir la valeur " . ($i + 1) . ": "));;
        if ($vals[$i] >= 0) {
            $pos++;
        } else {
            $neg++;
        }
    }
    
    echo "Nombre de valeurs positives : $pos\n";
    echo "Nombre de valeurs négatives : $neg\n\n";

Exercice 14


    $tab = array();
    $nbNotes = intval(readline("Combien de notes voulez-vous saisir ? "));
    $smNotes = 0;

    for ($i = 0; $i < $nbNotes; $i++) {
        $tab[$i] = intval(readline("Saisir la note " . ($i + 1) . ": "));
        $smNotes += $tab[$i];
    }

    $moyenne = $smNotes / $nbNotes;
    $moyenneSup = 0;

    for ($i = 0; $i < count($tab); $i++) {
        if ($tab[$i] > $moyenne) {
            $moyenneSup++;
        }
    }

    echo "La moyenne de la classe est : $moyenne\n";
    echo "Nombre de notes supérieures à la moyenne : $moyenneSup\n";

Console :


    Combien de notes voulez bous saisir ? 3
    Saisir la note 1 : 5
    Saisir la note 2 : 10
    Saisir la note 3 : 15
    La moyenne de la classe est 10
    Nombre de notes supérieur à la moyenne : 1
