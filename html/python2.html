    def liste(n):
        if n // 10==0:
            return [n]

    def somme(n):
        if len(liste(n)) == 1:
            return liste(n)[0] ** 2

    def heureux(n):
        try:
            if somme(n) == 1:
                return True
            else :
                print(n)

Algorithme de syracuse


    def syracuse(u):
        if u == 1:
            return [1]
        else:
            if u%2==0:
                u=u//2
            else:
                u=3*u+1
            return [u] + syracuse(u)
