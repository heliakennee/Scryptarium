        for ($i=0;$i<7;$i++) {
            $tab[] = 0;
        }

Exercice 5


    function fibre($tab) {
        for ($i=1;$i<6;$i++) {
            $tab[]= $tab[$i-1]+2;

        var_dump($tab);

Exercice 7


    function moyenne($tab) {
        $v = 0;
        $l = count($tab)
        for ($i=0;$i<$l;$i++) {
            $v+=$tab[$i];
        }
        return $v/count($tab);
    }
    echo moyenne(notes([5,10,15]));

Console :


    10
